# init
```bash
# yum 安装 git && 配置pub_key
yum install git

# 安装nvm
git clone git://github.com/creationix/nvm.git ~/nvm
echo "source ~/nvm/nvm.sh" >> ~/.bashrc
source ~/.bashrc

nvm list-remote
nvm install v14.0.0 # 安装node & npm
nvm alias default v14.0.0

# nrm
npm install -g nrm
nrm add <alias> https://npm.xx.com/
nrm use alias


# 安装yarn
npm install -g yarn
```
